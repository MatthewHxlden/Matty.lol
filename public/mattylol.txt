        // Custom Cursor Logic
        const cursor = document.getElementById('cursor');
        document.addEventListener('mousemove', (e) => {
            cursor.style.left = e.clientX + 'px';
            cursor.style.top = e.clientY + 'px';
        });

        document.querySelectorAll('a, .card').forEach(item => {
            item.addEventListener('mouseenter', () => {
                cursor.style.width = '40px';
                cursor.style.height = '40px';
                cursor.style.background = 'rgba(0, 243, 255, 0.1)';
            });
            item.addEventListener('mouseleave', () => {
                cursor.style.width = '20px';
                cursor.style.height = '20px';
                cursor.style.background = 'transparent';
            });
        });

        // Time logic
        function updateTime() {
            const now = new Date();
            document.getElementById('current-time').innerText = now.toLocaleString('en-US', { 
                weekday: 'short', 
                month: 'short', 
                day: 'numeric', 
                hour: '2-digit', 
                minute: '2-digit',
                second: '2-digit'
            });
        }
        setInterval(updateTime, 1000);
        updateTime();

        // Subtle 3D tilt effect on window
        const shell = document.querySelector('.mac-shell');
        document.addEventListener('mousemove', (e) => {
            let xAxis = (window.innerWidth / 2 - e.pageX) / 50;
            let yAxis = (window.innerHeight / 2 - e.pageY) / 50;
            shell.style.transform = `rotateY(${xAxis}deg) rotateX(${yAxis}deg)`;
        });
